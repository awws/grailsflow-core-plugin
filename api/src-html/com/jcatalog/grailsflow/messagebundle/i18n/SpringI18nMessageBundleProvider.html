001package com.jcatalog.grailsflow.messagebundle.i18n;
002
003import org.apache.commons.lang.ArrayUtils;
004import org.apache.commons.lang.StringUtils;
005import org.apache.commons.logging.Log;
006import org.apache.commons.logging.LogFactory;
007
008import org.springframework.beans.factory.InitializingBean;
009
010import org.springframework.context.ResourceLoaderAware;
011import org.springframework.context.support.ReloadableResourceBundleMessageSource;
012
013import org.springframework.core.io.ResourceLoader;
014
015import java.util.Iterator;
016import java.util.List;
017
018
019/**
020 * It extends {@link CachingI18nMessageBundleProvider} for caching already loaded
021 * message bundles. For loading a message bundle this implementation uses Spring
022 * <code>ReloadableResourceBundleMessageSource</code> as base - then it will be used for
023 * creating {@link SpringReloadableI18nMessageBundle}. Note, that {@link
024 * SpringI18nMessageBundleProvider#bundlesLocation} and {@link
025 * SpringI18nMessageBundleProvider#resourceLoader} should be appropriate each other,
026 * i.e. resorce loader should can to load message bundle with using its name and the
027 * specified base path. For more information see Soring
028 * <code>ReloadableResourceBundleMessageSource</code>.
029 *
030 * @author Roman Denisyuk
031 *
032 * @see CachingI18nMessageBundleProvider
033 * @see SpringReloadableI18nMessageBundle
034 */
035public class SpringI18nMessageBundleProvider extends CachingI18nMessageBundleProvider
036    implements ResourceLoaderAware, InitializingBean
037{
038    private static final Log log = LogFactory.getLog(SpringI18nMessageBundleProvider.class);
039
040    /* value of it is depends of what ResourceLoader we use */
041    private String bundlesLocation;
042    private List<String> bundlesLocations;
043    private int cacheSeconds = -1;
044    private ResourceLoader resourceLoader;
045
046    protected I18nMessageBundle loadMessageBundle(String bundleName) {
047        // create full to bundle
048        String[] bundlePathes = new String[] {  };
049        if (StringUtils.isNotEmpty(bundlesLocation)) {
050            bundlePathes = (String[]) ArrayUtils.add(bundlePathes,
051                    bundlesLocation + StringUtils.replace(bundleName, ".", "/"));
052        }
053        if (getBundlesLocations() != null) {
054            for (Iterator it = getBundlesLocations().iterator(); it.hasNext();) {
055                bundlePathes = (String[]) ArrayUtils.add(bundlePathes,
056                        it.next() + StringUtils.replace(bundleName, ".", "/"));
057            }
058        }
059        ReloadableResourceBundleMessageSource messageSource = new ReloadableResourceBundleMessageSource();
060        messageSource.setResourceLoader(resourceLoader);
061        messageSource.setBasenames(bundlePathes);
062        messageSource.setCacheSeconds(cacheSeconds);
063        messageSource.setFallbackToSystemLocale(false);
064        return new SpringReloadableI18nMessageBundle(messageSource);
065    }
066
067    public void setResourceLoader(ResourceLoader resourceLoader) {
068        this.resourceLoader = resourceLoader;
069    }
070
071    public void afterPropertiesSet() throws Exception {
072        if (resourceLoader == null) {
073            throw new IllegalStateException("ResourceLoader should be specified for "
074                + this.getClass().getName());
075        }
076        if (bundlesLocation == null) {
077            bundlesLocation = "";
078            log.warn(
079                "Property bundlesLocation isn't specified. Will be used empty value for it, but configured ResourcesLoader should can to load MessageBundle by its name.");
080        }
081    }
082
083    public void setBundlesLocation(String bundlesLocation) {
084        this.bundlesLocation = bundlesLocation;
085    }
086
087    public void setCacheSeconds(int cacheSeconds) {
088        this.cacheSeconds = cacheSeconds;
089    }
090
091    public List<String> getBundlesLocations() {
092        return bundlesLocations;
093    }
094
095    public void setBundlesLocations(List<String> bundlesLocations) {
096        this.bundlesLocations = bundlesLocations;
097    }
098}
